```c#
using System;

class Calc{
    public double CalcCircle(int rad = 1, double pi = 3.14){
        return rad * rad * pi;
    }
    public void Output(){
        double area;
        area = CalcCircle();
        Console.WriteLine(area);
        area = CalcCircle(10);
        Console.WriteLine(area);
        area = CalcCircle(20,3.14159);
        Console.WriteLine(area);
        area = CalcCircle(pi:3.14159,rad:20);
        Console.WriteLine(area);
    }
}
public class Hello{
    public static void Main(){
        // Your code here!
        Calc aa = new Calc();
        aa.Output();
    }
}
```
