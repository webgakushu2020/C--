using System;
using System.IO;
public class Hello{
    public static void Main(){
        FileStream fs;
        
        try{
            fs = new FileStream("file.txt",FileMode.Open);
        }
        catch (IOException){
            Console.WriteLine("ファイルを開けません。");
            return;
        }
            StreamReader r = new StreamReader(fs);
            
            string s;
            int i = 1;
            while((s = r.ReadLine()) != null){
                Console.WriteLine(i + ":" + s);
                i++;
            }
            r.Close();
    }
}
